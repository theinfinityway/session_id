## Examples
### Convert to Ed25519
```ts
import { Converter, ID } from "@li0ard/session_id"

let a = new Converter()
let id = new ID("05d871fc80ca007eed9b2f4df72853e2a2d5465a92fcb1889fb5c84aa2833b3b40")
console.log(a.convertToEd25519Key(id).toString(true))
```
